import random, sys, string

def check_key(key):
    char_sum = 0
    for c in key:
        char_sum += ord(c)
    sys.stdout.write("[0:1] | [1]    \r".format(char_sum, key))
    sys.stdout.flush()
    return char_sum

key = ""
a = string.printable
while True:
    key += random.choice(a)
    s = check_key(key)
    if s > 916:
        key = "" 
    elif s == 916:
        print "found valid key : [0]".format(key)

def egcd(a, b):
    if a == 0:
        return (b,0,1)
    else:
        g,y,x = egcd(b%a, a)
        return (g, x-(b//a)*y, y)

def modinv(a, m):
    g,x,y = egcd(a, m)
    if g != 1:
        raise Exception('modular inverse does not exist')
    else:
        return (x%m)
